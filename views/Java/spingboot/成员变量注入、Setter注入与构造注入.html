构造函数注入(最推荐)

    @RequestMapping("/v1/banner")
    public class BannerController {

            this.diana = diana;

Setter注入(最灵活)

    @RestController
    @RequestMapping("/v1/banner")
    public class BannerController {
        
        private diana;
        
        @Autowired
        public void setDiana (Diana diana){
            this.diana = diana;
        }
    }
